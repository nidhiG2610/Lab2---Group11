function drawStairs(n) { 
  var string = '';
  if(n>1){
     for(i=1;i<=n;i++){    
        if(i!=n){
          string += "I\n";
          string += ' '.repeat(i);
        }else{
          string += 'I';
        }      
    }     
      return  string;
  }else{
      return 'I';
  } 
}
